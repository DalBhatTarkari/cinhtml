 #include <stdio.h>
void main()
{
	int av=5;
    int a[av][av];
    int p[av][av*2+1];
    int i,j,x,y;
    for(i=0;i <av;i++)
    {
        for(j=0;j<av;j++)
        {
            //printf("Enter value for ROW %d and COLUMN %d: ",i+1,j+1);
            //scanf("%d",&a[i][j]);
            //printf("\n");
            a[i][j] = 0;
        }
        for(j=0;j<av*2-1;j++)
        {
            //printf("Enter value for ROW %d and COLUMN %d: ",i+1,j+1);
            //scanf("%d",&p[i][j]);
            //printf("\n");
            p[i][j] = 0;
        }
    }
    printf("\n");
    printf("SQUARE\n");
    for(i=0;i<av;i++)
    {
        for(j=0;j<av;j++)
        {
            printf("%d \t",a[i][j]);
        }
        printf("\n");
    }
    printf("\n");
    printf("RECTANGLE\n");
    for(i=0;i<av;i++)
    {
        for(j=0;j<av*2-1;j++)
        {
            printf("%d \t",p[i][j]);
        }
        printf("\n");
    }
    printf("\n");
    printf("RIGHT HALF PYRAMID\n");
    for(i=0;i<av;i++)
    {
        for(j=0;j<=i;j++)
        {
            printf("%d \t",a[i][j]);
        }
        printf("\n");
    }
    printf("\n");
    printf("INVERSE RIGHT HALF PYRAMID\n");
    for(i=0;i<av;i++)
    {
        for(j=av-1;j>=i;j--)
        {
            printf("%d \t",a[i][j]);
        }
        printf("\n");
    }
    printf("\n");
    printf("LEFT HALF PYRAMID\n");
    x = av;
    for (i = 0; i < av; i++) {

        for (j = 0; j < x - 1; j++)
		{
            printf("\t");
        }

        for (j = x - 1; j < av; j++)
		{
            printf("%d\t", a[i][j]);
        }
        x--;
        printf("\n");
    }
    printf("\n");
    printf("INVERSE LEFT HALF PYRAMID\n");
    x = 0;
    for (i = 0; i < av; i++)
	{
	    for(j = 0;j<x;j++)
        {
            printf("\t");
        }
        for (j = x ; j < av; j++)
		{
            printf("%d\t", a[i][j]);
        }
        printf("\n");
        x++;
    }
    printf("\n");
    printf("FULL PYRAMID\n");

    x = y = 4;

    for (i = 0; i < av; i++)
	{
	    for(j = 0;j<x;j++)
        {
            printf("\t");
        }
        for (j = x ; j <= y; j++)
	 {
            printf("%d\t", p[i][j]);
        }
        x--;
        y++;
        printf("\n");
    }
}
