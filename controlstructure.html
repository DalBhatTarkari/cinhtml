Switch Case Structure

Code:

#include 
int main() {
    char operator;
    float n1, n2;
    printf("Enter an operator: ");
    scanf("%c", &operator);
    printf("Enter a numbers: ");
    scanf("%f",&n1);
    printf("Enter anpther numbers: ");
    scanf("%f",&n2);
    switch(operator)
    {
        case '+':
            printf("%.1f + %.1f = %.1f",n1, n2, n1+n2);
            break;
        case '-':
            printf("%.1f - %.1f = %.1f",n1, n2, n1-n2);
            break;
        case '*':
            printf("%.1f * %.1f = %.1f",n1, n2, n1*n2);
            break;
        case '/':
            printf("%.1f / %.1f = %.1f",n1, n2, n1/n2);
            break;
        default:
            printf("Invalid Operator");
    }
    return 0;
}
